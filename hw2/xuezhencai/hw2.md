function ⊕(s::Set{Int}, t::Set{Int})
    return union(s, t)
end

function ⊙(s::Set{Int}, t::Set{Int})
    if isempty(t)
        return Set{Int}()
    else
        return Set([a * b for a in s for b in t])
    end
end
